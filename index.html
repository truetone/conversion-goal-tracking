Tracking Virtual Pageviews

// When someone clicks or tabs to the next form element,
// we record they've completed a step.                                                             
$('.new-group-reg').live('blur', function ()                                                                                                             
{                                                                                                                                                        
	if ($(this).val() != '')                                                                                                                             
    {                                                                                                                                                    
        pageview_string = '/groups/reg-tracking/new/' + $(this).attr('id');                                                                              
                                                                                                                                                             
        // Push the data to GA.                                                                                                                          
        _gaq.push(['_trackPageview', pageview_string]);                                                                                                  
    }                                                                                                                                                    
});
